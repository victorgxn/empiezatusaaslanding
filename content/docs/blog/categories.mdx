## Componente CategoryBadge

```typescript title="components/blog/category-badge.tsx"
import Link from 'next/link';

interface CategoryBadgeProps {
  category: string;
  size?: 'sm' | 'md' | 'lg';
  showIcon?: boolean;
}

export function CategoryBadge({
  category,
  size = 'md',
  showIcon = true,
}: CategoryBadgeProps) {

  const sizeClasses = {
    sm: 'text-xs px-2 py-0.5',
    md: 'text-sm px-3 py-1',
    lg: 'text-base px-4 py-2',
  };

  return (
    <Link
      href={`/blog/category/${category}`}
      className={`inline-flex items-center gap-1.5 rounded-full ${cat?.color || 'bg-primary'} text-white font-medium transition-opacity hover:opacity-80 ${sizeClasses[size]}`}
    >
      {showIcon && cat?.icon && <span>{cat.icon}</span>}
      <span>{cat?.name || category}</span>
    </Link>
  );
}
```

```typescript title="components/blog/tag-list.tsx"
import Link from 'next/link';

interface TagListProps {
  tags: string[];
  limit?: number;
  size?: 'sm' | 'md';
}

export function TagList({ tags, limit, size = 'sm' }: TagListProps) {
  const displayTags = limit ? tags.slice(0, limit) : tags;
  const remaining = limit ? tags.length - limit : 0;

  const sizeClasses = {
    sm: 'text-xs px-2 py-0.5',
    md: 'text-sm px-3 py-1',
  };

  return (
    <div className="flex flex-wrap gap-2">
      {displayTags.map((tag) => (
        <Link
          key={tag}
          href={`/blog/tag/${tag}`}
          className={`rounded-full bg-secondary text-secondary-foreground hover:bg-primary hover:text-primary-foreground transition-colors ${sizeClasses[size]}`}
        >
          #{tag}
        </Link>
      ))}
      {remaining > 0 && (
        <span className={`text-muted-foreground ${sizeClasses[size]}`}>
          +{remaining} m√°s
        </span>
      )}
    </div>
  );
}
```
